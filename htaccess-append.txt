
# This site doesn't use PHP, if something happens to land
# within the docroot this will defang it.
<Files ^(*.php|*.phps)>
    order deny,allow
    deny from all
</Files>

# Similarly defang other stuff
<FilesMatch "\.(php|pl|py|jsp|asp|htm|shtml|sh|cgi.+)$">
    ForceType text/plain
</FilesMatch>

# Disallow access to the .ht* files
<Files ~ "^.ht">
    Order allow,deny
    Deny from all
</Files>

RedirectMatch permanent ^/partials.html /layout/partials.html
